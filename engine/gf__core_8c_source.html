60gf_engine_t* gf_engine_create(const char* title, int nogui) {
61 int st;
62 gf_engine_t* engine = malloc(sizeof(*engine));
63 memset(engine, 0, sizeof(*engine));
64 engine->log = stderr;
65 engine->error = 0;
66 engine->lua = NULL;
67 if(nogui) {
68 gf_log_function(engine, "No GUI mode", "");
69 engine->client = NULL;
70 } else {
71 gf_log_function(engine, "GUI mode", "");
72 engine->client = gf_client_create(engine, title);
73 if(engine->client == NULL) {
74 gf_log_function(engine, "Failed to create client interface", "");
75 gf_engine_destroy(engine);
76 return NULL;
77 }
78 gf_log_function(engine, "Switching to graphical console", "");
79 }
80 engine->server = gf_server_create(engine);
81
82 engine->base = gf_resource_create(engine, "base.pak");
83 engine->lua = gf_lua_create(engine);
84 if((st = gf_lua_run(engine->lua, "base:/scripts/init.lua")) != 0) {
85 gf_assert(engine, st == 0);
86 gf_engine_destroy(engine);
87 engine = NULL;
88 }
89 return engine;
90}
