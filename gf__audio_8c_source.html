89gf_audio_id_t gf_audio_load(gf_audio_t* audio, const void* data, size_t size) {
90 gf_audio_decoder_t decoder;
91 int xm_cond;
92 int mod_cond;
93 int ind;
94
95 ma_mutex_lock(audio->mutex);
96
97 decoder.used = 0;
98 decoder.audio = audio;
99 decoder.decoder = NULL;
100 decoder.xm = NULL;
101 decoder.mod = NULL;
102 decoder.decoder_config = ma_decoder_config_init(audio->device_config.playback.format, audio->device_config.playback.channels, audio->device_config.sampleRate);
103 decoder.key = 0;
104 decoder.data = NULL;
105 do {
106 ind = hmgeti(audio->decoder, decoder.key);
107 if(ind != -1) {
108 decoder.key++;
109 }
110 } while(ind != -1);
111
112 xm_cond = size > 37 && memcmp(data, "Extended Module: ", 17) == 0 && ((char*)data)[37] == 0x1a;
113 mod_cond = size > 1080;
114
115 if(mod_cond) {
116 int j;
117 int mod_sig_cond = 0;
118 for(j = 0; j < sizeof(gf_audio_mod_sig) / sizeof(gf_audio_mod_sig[0]); j++) {
119 mod_sig_cond = mod_sig_cond || (memcmp((char*)data + 1080, gf_audio_mod_sig[j], 4) == 0);
120 }
121 mod_cond = mod_cond && mod_sig_cond;
122 }
123
124 if(xm_cond) {
125 if(jar_xm_create_context_safe(&decoder.xm, data, size, audio->device_config.sampleRate) == 0) {
126 /* In XM loader .internal is used to store old loopcount */
127 decoder.internal = jar_xm_get_loop_count(decoder.xm);
128 decoder.used = -1;
129 jar_xm_set_max_loop_count(decoder.xm, 1);
130 hmputs(audio->decoder, decoder);
131 ma_mutex_unlock(audio->mutex);
132 return decoder.key;
133 }
134 decoder.xm = NULL;
135 } else if(mod_cond) {
136 decoder.mod = malloc(sizeof(*decoder.mod));
137 jar_mod_init(decoder.mod);
138 decoder.mod->modfile = malloc(size);
139 decoder.mod->modfilesize = size;
140 memcpy(decoder.mod->modfile, data, size);
141 jar_mod_setcfg(decoder.mod, audio->device_config.sampleRate, 16, 1, 1, 0);
142 if(jar_mod_load(decoder.mod, (void*)decoder.mod->modfile, size)) {
143 /* In MOD loader .internal is used to store old loopcount */
144 decoder.internal = decoder.mod->loopcount;
145 decoder.used = -1;
146 hmputs(audio->decoder, decoder);
147 ma_mutex_unlock(audio->mutex);
148 return decoder.key;
149 }
150 free(decoder.mod->modfile);
151 free(decoder.mod);
152 decoder.mod = NULL;
153 }
154 decoder.decoder = malloc(sizeof(*decoder.decoder));
155 decoder.data = malloc(size);
156 decoder.size = size;
157 memcpy(decoder.data, data, size);
158 if(ma_decoder_init_memory(decoder.data, decoder.size, &decoder.decoder_config, decoder.decoder) == MA_SUCCESS) {
159 decoder.used = -1;
160 hmputs(audio->decoder, decoder);
161 ma_mutex_unlock(audio->mutex);
162 return decoder.key;
163 }
164 free(decoder.data);
165 free(decoder.decoder);
166 decoder.decoder = NULL;
167 ma_mutex_unlock(audio->mutex);
168 return -1;
