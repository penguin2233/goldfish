123int gf_draw_step(gf_draw_t* draw) {
124 int ret;
125 if(draw->fps != -1) {
126 gf_draw_time_t t;
127 gf_draw_time(&t);
128
129 draw->fps = 1.0 / (gf_draw_time_number(&t) - gf_draw_time_number(&draw->last_draw));
130 }
131 gf_draw_time(&draw->last_draw);
132 ret = gf_draw_platform_step(draw);
133 if(ret != 0) return ret;
134 if(draw->close == 1 && draw->engine->lua != NULL) {
135 draw->close = 0;
136 gf_lua_close(draw->engine->lua);
137 }
138 if(draw->fps == -1) {
139 draw->fps = 0;
140 } else {
141 double sfps = 60;
142 gf_draw_time_t t;
143
144 do {
145 gf_draw_time(&t);
146 } while((gf_draw_time_number(&t) - gf_draw_time_number(&draw->last_draw)) < 1.0 / sfps);
147 }
148
149 return draw->close;
150}
