136void gf_graphic_perspective(gf_draw_t* draw, double fovy, double znear, double zfar) {
137 double aspect = (double)draw->width / (double)draw->height;
138 double f = gf_math_cot(fovy / 180 * GF_MATH_PI / 2);
139 GLdouble matrix[16];
140 int i;
141
142 for(i = 0; i < 16; i++) matrix[i] = 0;
143 matrix[4 * 0 + 0] = f / aspect;
144 matrix[4 * 1 + 1] = f;
145 matrix[4 * 2 + 2] = (zfar + znear) / (znear - zfar);
146 matrix[4 * 3 + 2] = ((double)2 * zfar * znear) / (znear - zfar);
147 matrix[4 * 2 + 3] = -1;
148
149 glLoadIdentity();
150 glLoadMatrixd(matrix);
