38int gf_draw_platform_step(gf_draw_t* draw) {
39 int ret = 0;
40 int w, h;
41 RGFW_window_makeCurrent(draw->platform->window);
42 draw->close = RGFW_window_shouldClose(draw->platform->window);
43 while(RGFW_window_checkEvent(draw->platform->window)) {
44 if(draw->platform->window->event.type == RGFW_quit) {
45 draw->close = 1;
46 RGFW_window_setShouldClose(draw->platform->window, RGFW_FALSE);
47 break;
48 } else if(draw->platform->window->event.type == RGFW_mousePosChanged) {
49 draw->input->mouse_x = draw->platform->window->event.point.x;
50 draw->input->mouse_y = draw->platform->window->event.point.y;
51 } else if(draw->platform->window->event.type == RGFW_windowMoved) {
52 draw->x = draw->platform->window->r.x;
53 draw->y = draw->platform->window->r.y;
54 } else if(draw->platform->window->event.type == RGFW_windowResized) {
55 draw->width = draw->platform->window->r.w;
56 draw->height = draw->platform->window->r.h;
57 } else if(draw->platform->window->event.type == RGFW_mouseButtonPressed) {
58 if(draw->platform->window->event.button == RGFW_mouseLeft) draw->input->mouse_flag |= GF_INPUT_MOUSE_LEFT_MASK;
59 if(draw->platform->window->event.button == RGFW_mouseMiddle) draw->input->mouse_flag |= GF_INPUT_MOUSE_MIDDLE_MASK;
60 if(draw->platform->window->event.button == RGFW_mouseRight) draw->input->mouse_flag |= GF_INPUT_MOUSE_RIGHT_MASK;
61 } else if(draw->platform->window->event.type == RGFW_mouseButtonReleased) {
62 if(draw->platform->window->event.button == RGFW_mouseLeft) draw->input->mouse_flag ^= GF_INPUT_MOUSE_LEFT_MASK;
63 if(draw->platform->window->event.button == RGFW_mouseMiddle) draw->input->mouse_flag ^= GF_INPUT_MOUSE_MIDDLE_MASK;
64 if(draw->platform->window->event.button == RGFW_mouseRight) draw->input->mouse_flag ^= GF_INPUT_MOUSE_RIGHT_MASK;
65 }
66 }
67 if(ret == 0) {
68 gf_draw_driver_before(draw);
69 gf_draw_frame(draw);
70 gf_draw_driver_after(draw);
71
72 RGFW_window_swapBuffers(draw->platform->window);
73 }
74 return ret;
75}
